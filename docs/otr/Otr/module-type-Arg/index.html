Module type Otr.Arg

All user defined args conform to the module signature Arg.

For e.g.

module Fruit = struct
  type t =
    | Apple
    | Orange
    | Pineapple

  let t : t Otr.arg =
    Otr.create_arg ~name:"fruit" ~decode:(function
      | "apple" -> Some Apple
      | "orange" -> Some Orange
      | "pineapple" -> Some Pineapple
      | _ -> None)
end
type t
val t : t arg
